# GET


# Define key string
$key = "test10000"

# Construct the URL with the key parameter
$uri = "http://localhost:8091/get?key=$key"

# Send the GET request
$response = Invoke-RestMethod -Uri $uri -Method Get

# Print the response content
$response











# PUT (Expecting Two strings: Key Value)


$uri = "http://localhost:8092/put" 

# Define key and value strings
$key = "test10000"
$value = "*value*"

# Convert the key and value strings to JSON format
$jsonData = @"
{
    "key": "$key",
    "value": "$value"
}
"@

# Send the PUT request
$response = Invoke-RestMethod -Uri $uri -Method Put -ContentType "application/json" -Body $jsonData
